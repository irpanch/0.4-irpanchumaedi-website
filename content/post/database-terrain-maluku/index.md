function filterTabel(kata) {
  var rows = document.querySelectorAll("#tbody-tile tr");
  var kata_lower = kata.toLowerCase();
  var tampil = 0;
  rows.forEach(function(row) {
    var nlp = row.cells[0] ? row.cells[0].textContent.toLowerCase() : "";
    var sensor = row.cells[1] ? row.cells[1].textContent.toLowerCase() : "";
    if (nlp.includes(kata_lower) || sensor.includes(kata_lower)) {
      row.style.display = "";
      tampil++;
    } else {
      row.style.display = "none";
    }
  });
  var info = document.getElementById("info-hasil");
  if (kata !== "") {
    info.textContent = "Menampilkan " + tampil + " dari 309 tile";
  } else {
    info.textContent = "";
  }
}
